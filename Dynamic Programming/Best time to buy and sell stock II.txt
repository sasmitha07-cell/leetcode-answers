122. BEST TIME TO BUY AND SELL STOCK II


class Solution {
    public int maxProfit(int[] prices) {
        int n=prices.length;
        int[] hold=new int[n];
        int[] c=new int[n];
        hold[0]=-prices[0];
        c[0]=0;
        for(int i=1;i<n;i++){
            hold[i]=Math.max(hold[i-1],c[i-1]-prices[i]);
            c[i]=Math.max(c[i-1],hold[i-1]+prices[i]);
        }
        return c[n-1];
    }
}
