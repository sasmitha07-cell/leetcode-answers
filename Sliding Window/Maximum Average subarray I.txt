class Solution {
    public double findMaxAverage(int[] nums, int k) {
        double avg=0;
        double sum=0;
        for(int i=0;i<k;i++){
            sum+=nums[i];
        }
        avg=(sum/k);
        int j=0;
        for(int i=k;i<nums.length;i++){
            sum-=nums[j];
            sum+=nums[i];
            double a=(sum/k);
            avg=Math.max(a,avg);
            j++;
        }
        return avg;
    }
}
