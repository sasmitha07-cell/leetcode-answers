class Solution {
    public int numberOfSubstrings(String s) {
        int[] f=new int[3];
        int left=0;
        int c=0;
        for(int i=0;i<s.length();i++){
            f[s.charAt(i)-'a']++;
            while(f[0]>0 && f[1]>0 && f[2]>0){
                c+=s.length()-i;
                f[s.charAt(left)-'a']--;
                left++;
            }
        }
        return c;
    }
}
