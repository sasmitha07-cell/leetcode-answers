class Solution {
    public int divisorSubstrings(int num, int k) {
        int c=0;
        String a=String.valueOf(num);
        for(int i=0;i<=a.length()-k;i++){
            int val=Integer.parseInt(a.substring(i,i+k));
            if(val!=0 && num%val==0){
                c++;
            }
        }
        return c;
    }
}
