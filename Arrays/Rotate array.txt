189. Rotate Array

class Solution {
    public void rotate(int[] nums, int k) {
        int[] t=new int[nums.length];
        k=k%nums.length;
        for(int i=0;i<nums.length;i++){
            t[(i+k)%nums.length]=nums[i];
        }
        for(int i=0;i<t.length;i++){
            nums[i]=t[i];
        }
    }
}
