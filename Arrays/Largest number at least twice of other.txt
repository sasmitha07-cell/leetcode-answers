class Solution {
    public int dominantIndex(int[] nums) {
       int max=0;
       int mi=0;
       for(int i=0;i<nums.length;i++){
            if(max<nums[i]){
                max=nums[i];
                mi=i;
            }
       }
       int sm=0;
       for(int i=0;i<nums.length;i++){
        if(nums[i]!=max){
            sm=Math.max(nums[i],sm);
        }
       }
       if(sm*2<=max){
        return mi;
       }
       return -1;
    }
}
