1295. Find Numbers with Even Number of Digits

    class Solution {
    public int findNumbers(int[] nums) {
        int c=0;
        for(int i=0;i<nums.length;i++){
            int dc=0;
            int temp=nums[i];
            while(temp>0){
                dc++;
                temp/=10;
            }
            if(dc%2==0){
                c++;
            }
        }
        return c;
    }
}
