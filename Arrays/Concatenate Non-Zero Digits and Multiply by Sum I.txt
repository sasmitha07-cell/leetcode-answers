class Solution {
    public long sumAndMultiply(int n) {
        long x=0;
        long sum=0;
        while(n>0){
            if(n%10!=0){
                int d=n%10;
                x=x*10+d;
                sum+=d;
            }
            n/=10;
        }
        long ans=0;
        while(x>0){
            ans=ans*10+(x%10);
            x/=10;
        }
        return ans*sum;
    }
}
