169. Majority Element

class Solution {
    public int majorityElement(int[] nums) {
       int a=0;
       for(int i=0;i<nums.length;i++){
        int c=0;
        for(int j=i+1;j<nums.length;j++){
            if(nums[i]==nums[j]){
                c++;
            }
        }
        if(c>=(nums.length/2)){
            a=nums[i];
        }
       }
       return a;
    }
}
